investment = 100000
 
def sortino(x1, x2):
    a = investment * (temps[0][0] * x1 + temps[1][0] * x2)
    b = investment * (temps[0][1] * x1 + temps[1][1] * x2)
    c = investment * (temps[0][2] * x1 + temps[1][2] * x2)
    d = investment * (temps[0][3] * x1 + temps[1][3] * x2)

    mar = 0.05

    q4 = (e / d - 1) - mar
 
    porttemp = []
    qq = (q1, q2, q3, q4)
